>添加自定义双拼方案  
>一个声母、韵母可以对应一个或多个按键；一对按键(声母+韵母按键)不能直接对应多个拼音（部分可通过添加模糊音规则实现），示例模版：
```
customspjm:={mmsp:{"0": "1"    ; mmsp 双拼标识不可重复，sp结尾 "0" 零声母模式
; "1" 两字母的韵母同全拼，a、o、e为aa、oo、ee，三字母韵母为韵母首字母+韵母对应键
; "2" 韵母首字母+韵母对应键
; "a"-"z" 固定零声母
            
; 双拼显示中文名
	,"1": "某某双拼"

; ----例外音节-----------不需要可省去-----------
; （不完全符合下述声母韵母按键关系的拼音）
	,"2":{"xiong": "fs"
	    , "shang": "hs"}
; --------------------------------------------

; ----模糊音正则规则-----不需要可省去-----------
; v结尾的拼音与i结尾的拼音单向模糊; 双向模糊 ["[iv]'", "[iv]'"]
    ,"3":[["v'", "[iv]'"]
; ju、qu、xu、yu与ji、qi、xi、yi单向模糊
	, ["([jqxy])u'", "$1[iu]'"]
; uan结尾的拼音与e结尾的拼音单向模糊
	, ["uan'", "(e|uan)'"]]
; --------------------------------------------

; zh、ch、sh按键映射，其他声母修改通过例外音节定义
	, "ch"	: "i"
	, "sh"	: "u"
	, "zh"	: "v"

; 韵母按键映射
	, "ai"	: "d"
	, "an"	: "j"
	, "ang"	: "h"
	, "ao"	: "c"
```
